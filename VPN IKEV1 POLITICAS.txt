! ============================================================
! ROUTER A — VPN POLICY-BASED IKEv1
! WAN: 200.0.0.1/30  LAN: 192.168.1.0/24
! ============================================================

! --- Interfaces ---
interface GigabitEthernet0/0
 description LAN - Site A
 ip address 192.168.1.1 255.255.255.0
 no shutdown
!
interface GigabitEthernet0/1
 description WAN hacia ISP
 ip address 200.0.0.1 255.255.255.252
 no shutdown

! --- Ruta por defecto hacia ISP ---
ip route 0.0.0.0 0.0.0.0 200.0.0.2

! --- FASE 1: ISAKMP Policy (IKEv1) ---
crypto isakmp policy 10
 encr aes 256
 hash sha256
 authentication pre-share
 group 14
 lifetime 86400

! --- Pre-Shared Key ---
crypto isakmp key VPN_S3CR3T address 200.0.0.5

! --- FASE 2: IPSec Transform Set ---
crypto ipsec transform-set TS_AES256_SHA256 esp-aes 256 esp-sha256-hmac
 mode tunnel

! --- Crypto ACL (tráfico a cifrar) ---
ip access-list extended ACL_VPN_A_TO_B
 permit ip 192.168.1.0 0.0.0.255 192.168.2.0 0.0.0.255

! --- Crypto Map ---
crypto map CMAP_SITE_A 10 ipsec-isakmp
 set peer 200.0.0.5
 set transform-set TS_AES256_SHA256
 set pfs group14
 match address ACL_VPN_A_TO_B

! --- Aplicar Crypto Map a interfaz WAN ---
interface GigabitEthernet0/1
 crypto map CMAP_SITE_A

Router B (Cisco IOS / IOS-XE)
